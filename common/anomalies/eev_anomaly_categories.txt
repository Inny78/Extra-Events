# Supernova
supernova_category = {
	desc = "supernova_category_desc"
	picture = GFX_evt_ship_in_orbit
	max_once_global = yes
	level = 3
	null_spawn_chance = 0.6
	spawn_chance = {
		modifier = {
			add = 1
			mid_game_years_passed < 0
			AND = {
				OR = {
					is_planet_class = pc_a_star
					is_planet_class = pc_b_star
					is_planet_class = pc_m_giant_star
					# Real Space
					is_planet_class_pc_a_super_star = yes
					is_planet_class_pc_k_giant_star = yes
					is_planet_class_pc_g_giant_star = yes
					is_planet_class_pc_o_star = yes
					is_planet_class_pc_o_super_star = yes
					is_planet_class_pc_o_hyper_star = yes
					# Gigastructural Engineering & More
					giga_is_superpower_star = yes
				}
				is_binary_star = no
				is_trinary_star = no
				eev_is_valid_system = yes
			}
		}
	}
	on_success = {
		0 = {
			modifier = {
				add = 1
				has_nemesis = yes
				has_global_flag = eev_kaboom
			}
			anomaly_event = supernova.1
		}
		0 = {
			modifier = {
				add = 1
				has_nemesis = no
				has_global_flag = eev_kaboom
			}
			anomaly_event = supernova.4
		}
		3 = {
			anomaly_event = supernova.5
		}
	}
	on_spawn = {
		clear_deposits = yes
	}
}
