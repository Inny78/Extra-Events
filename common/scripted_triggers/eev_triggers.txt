eev_mod_active = { always = yes }

eev_is_general_planet = {
	OR = {
		habitable_planet = yes
		eev_is_uninhabitable = yes
	}
}

eev_is_continental = {
	is_wet = yes

		is_planet_class = pc_continental

		atw_is_continental = yes
	}
}

eev_is_uninhabitable = {
	OR = {
		uninhabitable_regular_planet = yes
		eev_is_frozen = yes
		eev_is_barren = yes
		eev_is_molten = yes
		eev_is_barren_cold = yes
		# Planetary Diversity
		is_pd_toxic = yes
		# All These Worlds
		atw_is_toxic = yes
	}
}

eev_is_valid_system = {
	solar_system = {
		exists = star
		NOR = {
			any_system_planet = {
				OR = {
					exists = archaeological_site
					has_planet_flag = hatchling_egg
					has_planet_flag = suppress_archaeological_sites
				}
			}
			has_special_star_flag_trigger = yes
			has_star_flag = empire_cluster
			is_fe_cluster = yes
			has_star_flag = drone_home_system
		}
	}
}

# Listing habitats
eev_is_habitat = {
	OR = {
		is_planet_class = pc_habitat
		uses_district_set = habitat
		# Acquisition of Technology
		is_aot_habitat = yes
		# DarkSpace
		is_ds_skycatcher = yes
		# Gigastructural Engineering & More
		is_giga_habitat = yes
		# Habitation
		has_planet_flag = hab_init
		# Living Space
		has_planet_flag = stpg_planetary_habitat
		# NSC2 Modjam Freebooters Origin
		uses_district_set = nsc_freebooter_habitat
		# Twinks Atlantean
		uses_district_set = skypalace
		# ZOFE
		uses_district_set = utopian_habitat
	}
}

eev_is_valid_moon = {
	AND = {
		eev_is_barren = yes
		is_moon = yes
		NOT = { has_planet_flag = has_megastructure }
		solar_system = {
			starting_system = no
		}
	}
}

eev_is_valid_habitat = {
	AND = {
		eev_is_habitat = yes
		is_capital = no
		num_pops > 10
	}
}

eev_is_valid_colony = {
